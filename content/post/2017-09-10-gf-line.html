R4DS の以下を mosaic を使って再現する.

http://r4ds.had.co.nz/model-basics.html#formulas-and-model-families

library(tidyverse)
library(mosaic)
library(modelr)

Categorical variables

sim2 %>% gf_point(y~x)

mod2 <- lm(y ~ x, data = sim2)
grid <- sim2 %>% 
  data_grid(x) %>% 
  add_predictions(mod2)
sim2 %>% gf_point(y~x) %>% 
  gf_point(data = grid, pred~x, color ="red", size =4)
