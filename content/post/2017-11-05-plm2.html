vcovHCC <- function(fm) {
  G <- length(unique(Fatalities$state))
  N <- length(Fatalities$state)
  dfa <-(G/(G - 1)) * (N - 1)/(fm$df.residual+G)
  return(dfa*vcovHC(fm, cluster="group", type="HC0"))
}

timeeffect <- function(fm) {
  fmb <- update(fm, effect = "individual")
  fmc <- update(fmb, .~.+factor(year))
  waldtest(fmb,fmc,vcov=vcovHCC(fmc),test="F")
}
